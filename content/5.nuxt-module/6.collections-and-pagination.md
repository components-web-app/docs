<!--
COVER:
- Use case: a component backed by the Collection built-in resource (see API → Collection Component)
  Renders a list of API resources with pagination controls

- useCwaCollectionResource usage:
    import { useCwaCollectionResource } from '#imports'
    const props = defineProps<IriProp>()
    const {
        resource,
        getResource,
        collectionItems,
        isLoadingCollection,
        totalPages,
        pageModel,
        goToNextPage,
        goToPreviousPage,
        changePage
    } = useCwaCollectionResource(props)
    defineExpose(getResource.exposeMeta)

- collectionItems: the hydra:member array — each item is a full resource object
- isLoadingCollection: true while the proxied collection fetch is in progress
- totalPages: computed from hydra:totalItems / perPage
- pageModel: a v-model-compatible ref (updates the page query param in the URL)

- Minimal template:
    <div v-if="!isLoadingCollection">
        <div v-for="item in collectionItems" :key="item['@id']">
            {{ item.data?.headline }}
        </div>
        <div class="pagination">
            <button @click="goToPreviousPage" :disabled="pageModel.value <= 1">Previous</button>
            <span>{{ pageModel.value }} / {{ totalPages }}</span>
            <button @click="goToNextPage" :disabled="pageModel.value >= totalPages">Next</button>
        </div>
    </div>

- useCwaCollectionPagination: for rendering a numbered page list:
    import { useCwaCollectionPagination } from '#imports'
    const { pages } = useCwaCollectionPagination(totalPages, pageModel)
    — pages: computed array of page numbers with ellipsis markers for long ranges

- Linking to a resource's page (resolveResourceLink):
    import { resolveResourceLink } from '@cwa/nuxt/runtime/utils'
    const href = resolveResourceLink(item)
    — looks up the Route whose page data matches this resource's IRI
    — returns the path string for <NuxtLink :to="href">

- useQueryBoundModel — bind a filter or sort param to the URL query string:
    import { useQueryBoundModel } from '#imports'
    const sortModel = useQueryBoundModel('order[createdAt]', 'desc')
    — sortModel.value is reactive; changing it updates the URL and triggers a re-fetch
    — Use with a <select> for user-facing sort/filter controls

- Ordering in the collection: set in the API via defaultQueryParameters
    — client-side override via useQueryBoundModel

- Loading states: show skeleton or spinner while isLoadingCollection is true
  — collectionItems is an empty array (not null) while loading, so v-for won't error
-->
