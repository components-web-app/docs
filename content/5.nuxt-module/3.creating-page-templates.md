<!--
COVER:
- File convention:
    app/cwa/pages/Primary.vue → registered as CwaPagePrimary
    File name (without .vue) = uiComponent value in the API

- The IriProp type — every page component receives :iri:
    import type { IriProp } from '@cwa/nuxt/runtime/composables'
    defineProps<IriProp>()

- Minimal page template:
    <template>
        <main>
            <CwaComponentGroup reference="content" :location="iri" />
        </main>
    </template>
    <script setup lang="ts">
    import type { IriProp } from '@cwa/nuxt/runtime/composables'
    defineProps<IriProp>()
    </script>

- Multiple regions (hero + main + sidebar):
    <section class="hero">
        <CwaComponentGroup reference="hero" :location="iri" />
    </section>
    <div class="grid grid-cols-3">
        <CwaComponentGroup reference="main" :location="iri" class="col-span-2" />
        <CwaComponentGroup reference="sidebar" :location="iri" />
    </div>

- Accessing the page resource:
    const page = $cwa.resources.page.value
    const uiClassNames = page?.data?.uiClassNames

- Applying uiClassNames:
    <div :class="page?.data?.uiClassNames">...</div>
    — CSS classes registered in nuxt.config pages.[Name].classes

- Dynamic (template) pages: accessing page data:
    const pageData = $cwa.resources.pageData.value
    const pageDataIri = $cwa.resources.pageDataIri.value
    — pageData.data.title auto-overrides the <title> tag for SEO
    — pageData.data.metaDescription auto-set as the meta description

- nuxt.config pages registration:
    cwa: {
        pages: {
            Primary: { name: 'Primary', classes: [...] }
        }
    }

- Loading state: $cwa.resources.pageLoading.value
-->
